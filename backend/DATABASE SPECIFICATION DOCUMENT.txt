DATABASE SPECIFICATION DOCUMENT (DSD)
I. GIỚI THIỆU
1.1. Mục đích tài liệu
Database Specification Document đặc tả chi tiết cấu trúc vật lý của cơ sở dữ liệu được triển khai trên PostgreSQL . Tài liệu xác định:
Danh sách bảng và cột dữ liệu.
Kiểu dữ liệu PostgreSQL .
Khóa chính, khóa ngoại và khóa duy nhất.
Giá trị mặc định và khả năng để trống.
Miền giá trị và ràng buộc kiểm tra.
Chỉ mục phục vụ tìm kiếm.
Transaction phục vụ các nghiệp vụ quan trọng.
Chính sách xóa, bảo mật, sao lưu và phục hồi.
Khả năng truy vết từ yêu cầu đến bảng dữ liệu.
Cấu trúc cơ sở dữ liệu được suy diễn từ BRD, SyRS, Static Model và Dynamic Model. Static Model xác định 12 lớp <<entity>>, thuộc tính, quan hệ và bội số; Dynamic Model xác định trạng thái, quy tắc chuyển trạng thái và cách cập nhật dữ liệu trong từng nghiệp vụ.
1.2. Phạm vi
DSD bao phủ dữ liệu của các nhóm chức năng:
SyRS xác định hệ thống không bao gồm thanh toán trực tuyến, vận chuyển, theo dõi vị trí thời gian thực và tích hợp hệ thống bên ngoài; vì vậy DSD không thiết kế các bảng cho những chức năng này.
1.3. Tài liệu tham khảo
II. MÔI TRƯỜNG VÀ QUY ƯỚC TRIỂN KHAI
2.1. Cấu hình cơ sở dữ liệu
PostgreSQL hỗ trợ kiểu dữ liệu BOOLEAN nguyên bản, kiểu UUID, dữ liệu JSON/JSONB và các kiểu ngày giờ có múi giờ. Trong phạm vi đồ án, khóa chính được lưu bằng kiểu UUID và được sinh tự động bằng hàm gen_random_uuid() thuộc extension pgcrypto.
2.2. Khởi tạo database

CREATE TABLE nguoi_dung (
    nguoi_dung_id UUID NOT NULL DEFAULT gen_random_uuid(),
    email VARCHAR(254),
    so_dien_thoai VARCHAR(20),
    mat_khau_hash VARCHAR(255) NOT NULL,
    vai_tro VARCHAR(30) NOT NULL
        DEFAULT 'THANH_VIEN',
    trang_thai VARCHAR(30) NOT NULL
        DEFAULT 'CHO_XAC_THUC',
    ngay_tao TIMESTAMPTZ NOT NULL
        DEFAULT CURRENT_TIMESTAMP,
    ngay_cap_nhat TIMESTAMPTZ NOT NULL
        DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT pk_nguoi_dung
        PRIMARY KEY (nguoi_dung_id),
    CONSTRAINT uq_nguoi_dung_email
        UNIQUE(email),
    CONSTRAINT uq_nguoi_dung_so_dien_thoai
        UNIQUE(so_dien_thoai),
    CONSTRAINT ck_nguoi_dung_dang_nhap
        CHECK(
            email IS NOT NULL
            OR so_dien_thoai IS NOT NULL
        ),
    CONSTRAINT ck_nguoi_dung_vai_tro
        CHECK(
            vai_tro IN(
                'THANH_VIEN',
                'QUAN_TRI_VIEN'
            )
        ),
    CONSTRAINT ck_nguoi_dung_trang_thai
        CHECK(
            trang_thai IN(
                'CHO_XAC_THUC',
                'HOAT_DONG',
                'BI_KHOA',
                'NGUNG_HOAT_DONG'
            )
        )
);

CREATE TABLE bai_dang_tai_san(
    bai_dang_id UUID
        DEFAULT gen_random_uuid(),
    chu_so_huu_id UUID NOT NULL,
    danh_muc_id UUID NOT NULL,
    ten_tai_san VARCHAR(200)
        NOT NULL,
    mo_ta_hien_trang TEXT
        NOT NULL,
    so_luong_tong INTEGER
        NOT NULL,
    so_luong_kha_dung INTEGER
        NOT NULL,
    so_luong_giu_cho INTEGER
        DEFAULT 0,
    so_luong_da_phan_phoi INTEGER
        DEFAULT 0,
    dia_diem VARCHAR(255)
        NOT NULL,
    hinh_thuc_chia_se VARCHAR(30)
        NOT NULL,
    hinh_thuc_trao_doi VARCHAR(30),
    trang_thai VARCHAR(30)
        DEFAULT 'KHA_DUNG',
    ngay_tao TIMESTAMPTZ
        DEFAULT CURRENT_TIMESTAMP,
    ngay_cap_nhat TIMESTAMPTZ
        DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT pk_bai_dang
        PRIMARY KEY(bai_dang_id),
    CONSTRAINT fk_bai_dang_nguoi_dung
        FOREIGN KEY(chu_so_huu_id)
        REFERENCES nguoi_dung(nguoi_dung_id),
    CONSTRAINT fk_bai_dang_danh_muc
        FOREIGN KEY(danh_muc_id)
        REFERENCES danh_muc_tai_san(danh_muc_id),
    CONSTRAINT ck_so_luong
        CHECK(
            so_luong_tong>=1
            AND so_luong_kha_dung>=0
            AND so_luong_giu_cho>=0
            AND so_luong_da_phan_phoi>=0
            AND
            so_luong_kha_dung
            +so_luong_giu_cho
            +so_luong_da_phan_phoi
            =so_luong_tong
        ),
    CONSTRAINT ck_hinh_thuc_chia_se
        CHECK(
            hinh_thuc_chia_se
            IN(
                'CHO_TANG',
                'TRAO_DOI'
            )
        ),
    CONSTRAINT ck_hinh_thuc_trao_doi
        CHECK(
            (
                hinh_thuc_chia_se='CHO_TANG'
                AND hinh_thuc_trao_doi IS NULL
            )
            OR
            (
                hinh_thuc_chia_se='TRAO_DOI'
                AND hinh_thuc_trao_doi
                IN(
                    'TAI_SAN',
                    'TIEN'
                )
            )
        ),
    CONSTRAINT ck_trang_thai_bai_dang
        CHECK(
            trang_thai
            IN(
                'KHA_DUNG',
                'TAM_HET_KHA_DUNG',
                'DA_KET_THUC'
            )
        )
);

CREATE TABLE giao_dich (
    giao_dich_id UUID NOT NULL
        DEFAULT gen_random_uuid(),
    de_xuat_id UUID NOT NULL,
    nguoi_so_huu_id UUID NOT NULL,
    nguoi_tiep_nhan_id UUID NOT NULL,
    so_luong_giao_dich INTEGER NOT NULL,
    thoi_gian_ban_giao TIMESTAMPTZ NOT NULL,
    dia_diem_ban_giao VARCHAR(255) NOT NULL,
    xac_nhan_nguoi_so_huu BOOLEAN NOT NULL
        DEFAULT FALSE,
    xac_nhan_nguoi_tiep_nhan BOOLEAN NOT NULL
        DEFAULT FALSE,
    ly_do_huy TEXT,
    trang_thai VARCHAR(40) NOT NULL
        DEFAULT 'CHO_BAN_GIAO',
    ngay_tao TIMESTAMPTZ NOT NULL
        DEFAULT CURRENT_TIMESTAMP,
    ngay_cap_nhat TIMESTAMPTZ NOT NULL
        DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT pk_giao_dich
        PRIMARY KEY (giao_dich_id),
    CONSTRAINT uq_giao_dich_de_xuat
        UNIQUE (de_xuat_id),
    CONSTRAINT ck_giao_dich_so_luong
        CHECK (
            so_luong_giao_dich > 0
        ),
    CONSTRAINT ck_giao_dich_hai_ben
        CHECK (
            nguoi_so_huu_id <> nguoi_tiep_nhan_id
        ),
    CONSTRAINT ck_giao_dich_dia_diem
        CHECK (
            CHAR_LENGTH(TRIM(dia_diem_ban_giao)) > 0
        ),
    CONSTRAINT ck_giao_dich_trang_thai
        CHECK (
            trang_thai IN (
                'CHO_BAN_GIAO',
                'CHO_BEN_CON_LAI_XAC_NHAN',
                'HOAN_TAT',
                'DA_HUY'
            )
        ),
    CONSTRAINT ck_giao_dich_hoan_tat
        CHECK (
            trang_thai <> 'HOAN_TAT'
            OR (
                xac_nhan_nguoi_so_huu = TRUE
                AND xac_nhan_nguoi_tiep_nhan = TRUE
            )
        ),
    CONSTRAINT ck_giao_dich_huy
        CHECK (
            trang_thai <> 'DA_HUY'
            OR (
                ly_do_huy IS NOT NULL
                AND CHAR_LENGTH(TRIM(ly_do_huy)) > 0
            )
        ),
    CONSTRAINT fk_giao_dich_de_xuat
        FOREIGN KEY (de_xuat_id)
        REFERENCES de_xuat_giao_dich (de_xuat_id)
        ON UPDATE RESTRICT
        ON DELETE RESTRICT,
    CONSTRAINT fk_giao_dich_nguoi_so_huu
        FOREIGN KEY (nguoi_so_huu_id)
        REFERENCES nguoi_dung (nguoi_dung_id)
        ON UPDATE RESTRICT
        ON DELETE RESTRICT,
    CONSTRAINT fk_giao_dich_nguoi_tiep_nhan
        FOREIGN KEY (nguoi_tiep_nhan_id)
        REFERENCES nguoi_dung (nguoi_dung_id)
        ON UPDATE RESTRICT
        ON DELETE RESTRICT
);
